GET /users

Retrieve User objects. User objects are sorted based on the createdAt field in descending order.

limit

Number of User objects to return.

type
integer , { x ∈ ℤ | 0 ≤ x ≤ 1000 }
in
query
skip

Number of User objects to skip.

type
integer , { x ∈ ℤ | 0 ≤ x ≤ 1000 }
in
query
fields

List for fields to return. (Example) username,createdAt,roles

type
string
in
query
200 OK

List of User objects.

Response Example (200 OK) 
{
  "properties": {
    "data": {
      "items": {
        "properties": {
          "_id": {
            "$ref": "#/definitions/id"
          },
          "createdAt": {
            "format": "date-time",
            "type": "string"
          },
          "roles": {
            "items": {
              "$ref": "#/definitions/id"
            },
            "type": "array"
          },
          "uri": {
            "pattern": "^https?:\\/\\/.+\\/users/[a-f\\d]{24}$",
            "type": "string"
          },
          "username": {
            "$ref": "#/definitions/username"
          }
        },
        "required": [
          "_id",
          "uri"
        ],
        "type": "object"
      },
      "type": "array"
    },
    "links": {
      "properties": {
        "last": {
          "pattern": "^https?:\\/\\/.+\\/users\\?limit=\\d+&skip=\\d+$",
          "type": "string"
        },
        "next": {
          "pattern": "^https?:\\/\\/.+\\/users\\?limit=\\d+&skip=\\d+$",
          "type": "string"
        },
        "self": {
          "pattern": "^https?:\\/\\/.+\\/users\\?limit=\\d+&skip=\\d+$",
          "type": "string"

    }
  },
  "type": "object"
}
